### 2. 实现 isDuplicateMessage 函数

**位置**：Line 181-210

```javascript
/**
 * 检查消息是否重复
 * @param {object} message - 消息对象
 * @returns {boolean} - 是否重复
 */
const isDuplicateMessage = (message) => {
  // 优先使用服务端 _id，其次使用 clientMsgId
  const msgId = message._id || message.id || message.clientMsgId
  if (!msgId) {
    console.warn('⚠️ 消息缺少 ID，无法去重:', message)
    return false
  }
  
  if (receivedMsgIds.has(msgId)) {
    console.log('⚠️ 检测到重复消息，已跳过:', msgId)
    return true
  }
  
  receivedMsgIds.add(msgId)
  
  // 限制 Set 大小，避免内存泄漏
  if (receivedMsgIds.size > MAX_STORED_IDS) {
    // 清理最早的 500 个 ID
    const idsArray = Array.from(receivedMsgIds)
    idsArray.slice(0, 500).forEach(id => receivedMsgIds.delete(id))
    console.log('🗑️ 清理了 500 个旧消息 ID，当前大小:', receivedMsgIds.size)
  }
  
  return false
}
```

**功能说明**：
- ✅ 优先使用 `_id`（服务端生成，最可靠）
- ✅ 其次使用 `id` 或 `clientMsgId`（客户端生成）
- ✅ 检测到重复时返回 `true`，并记录日志
- ✅ 自动清理旧 ID，防止内存泄漏（最多存储 1000 个）

### 3. 修改 addMessageToConversation 函数

**位置**：Line 212-227

```javascript
/**
 * 添加消息到对话数组（带去重检查）
 */
const addMessageToConversation = (conversationId, message) => {
  // ✅ 检查是否重复
  if (isDuplicateMessage(message)) {
    return  // 重复消息，直接返回
  }
  
  // 初始化对话数组
  if (!conversations[conversationId]) {
    conversations[conversationId] = []
  }
  
  // 使用 splice 保持响应式
  conversations[conversationId].splice(
    conversations[conversationId].length,
    0,
    message
  )
}
```

**关键改进**：
- ✅ 在添加消息前先检查是否重复
- ✅ 如果重复，直接返回，不添加到数组
- ✅ 保持原有的响应式更新逻辑（使用 `splice`）
