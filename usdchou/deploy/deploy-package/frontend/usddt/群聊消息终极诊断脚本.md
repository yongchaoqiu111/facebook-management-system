## 📋 完整诊断脚本

```javascript
(async function ultimateDiagnosis() {
  console.log('\n🔍 ===== 群聊消息终极诊断开始 =====\n');
  
  // ==================== 第1部分: 检查 Socket 状态 ====================
  console.group('📌 第1部分: Socket 连接状态');
  
  try {
    const module = await import('/src/socket.js');
    const socket = module.getSocket();
    
    console.log('1. Socket 连接状态:', socket.connected ? '✅ 已连接' : '❌ 未连接');
    console.log('2. Socket ID:', socket.id || '未知');
    console.log('3. 当前房间列表:', Array.from(socket.rooms));
    
    // 检查是否加入了六合天下群
    const LIUHE_GROUP_ID = '69d3f89668f596338b0c1930';
    const hasGroup = Array.from(socket.rooms).some(r => 
      r.includes(LIUHE_GROUP_ID) || 
      r.includes(`group:${LIUHE_GROUP_ID}`)
    );
    
    console.log('4. 已加入六合天下群:', hasGroup ? '✅ 是' : '❌ 否');
    
    if (!hasGroup) {
      console.warn('⚠️ 警告: 未加入群组房间！正在尝试加入...');
      module.joinGroup(LIUHE_GROUP_ID);
      console.log('✅ 已调用 joinGroup，请刷新页面后重试');
    }
    
  } catch (error) {
    console.error('❌ Socket 模块加载失败:', error.message);
  }
  
  console.groupEnd();
  
  // ==================== 第2部分: 强制注册监听器 ====================
  console.group('📌 第2部分: 消息监听器测试');
  
  try {
    const module = await import('/src/socket.js');
    const socket = module.getSocket();
    
    // 移除可能存在的旧监听器
    socket.off('groupMessage');
    
    // 注册新的强制监听器
    socket.on('groupMessage', (data) => {
      console.log('\n🚨 ===== 强制捕获到群消息 =====');
      console.log('消息内容:', data);
      console.log('发送者:', data.sender);
      console.log('群组ID:', data.groupId);
      console.log('时间戳:', new Date().toLocaleTimeString());
      console.log('=====================================\n');
      
      // 检查数据格式
      if (!data.content) {
        console.warn('⚠️ 警告: 消息缺少 content 字段');
      }
      if (!data.sender) {
        console.warn('⚠️ 警告: 消息缺少 sender 字段');
      }
    });
    
    console.log('✅ 已注册强制监听器: socket.on("groupMessage")');
    console.log('💡 现在请在另一个窗口发送一条群聊消息');
    console.log('💡 如果看到 "🚨 强制捕获到群消息"，说明前端监听正常\n');
    
  } catch (error) {
    console.error('❌ 监听器注册失败:', error.message);
  }
  
  console.groupEnd();
  
  // ==================== 第3部分: 手动模拟测试 ====================
  console.group('📌 第3部分: 手动模拟消息测试');
  
  try {
    const module = await import('/src/socket.js');
    const socket = module.getSocket();
    
    console.log('正在发送测试消息...');
    
    // 手动触发一次本地测试
    setTimeout(() => {
      socket.emit('groupMessage', {
        _id: 'test_msg_' + Date.now(),
        groupId: '69d3f89668f596338b0c1930',
        content: '这是一条来自前端的测试推送',
        sender: { 
          _id: 'test_user', 
          userId: 'test_user',
          username: 'TestUser' 
        },
        clientMsgId: 'test_' + Date.now(),
        createdAt: new Date().toISOString()
      });
      
      console.log('✅ 测试消息已发送到后端');
      console.log('💡 如果后端正确广播，你应该会收到自己的消息\n');
    }, 1000);
    
  } catch (error) {
    console.error('❌ 测试消息发送失败:', error.message);
  }
  
  console.groupEnd();
  
  // ==================== 第4部分: 检查 UI 更新逻辑 ====================
  console.group('📌 第4部分: UI 更新检查');
  
  console.log('💡 如果收到了消息但界面没更新，可能是以下原因:');
  console.log('  1. onGroupMessage 回调中没有调用 addMessage()');
  console.log('  2. messages 数组没有正确响应式更新');
  console.log('  3. 消息被过滤掉了（isCurrentChat 判断失败）');
  console.log('\n💡 请检查 Chat.vue 中的 onGroupMessage 监听器代码');
  
  console.groupEnd();
  
  // ==================== 总结 ====================
  console.log('\n✅ ===== 诊断完成 =====\n');
  console.log('📝 下一步操作:');
  console.log('  1. 查看上面的诊断结果');
  console.log('  2. 在另一个窗口发送一条群聊消息');
  console.log('  3. 观察控制台是否有 "🚨 强制捕获到群消息"');
  console.log('  4. 如果有日志但界面没更新 → UI 问题');
  console.log('  5. 如果完全没有日志 → 后端推送问题\n');
  
})();
```

## 🔍 诊断结果解读

### 情况A: 看到 "🚨 强制捕获到群消息" ✅

**说明**: 
- ✅ 前端监听器工作正常
- ✅ 后端成功推送了消息
- ❌ **问题出在 UI 更新逻辑**

**解决方案**:
检查 `Chat.vue` 中的 `onGroupMessage` 回调，确保：
```javascript
onGroupMessage((data) => {
  console.log('收到群聊消息:', data)  // ← 这行应该有
  
  // ... 处理逻辑
  
  addMessage(message)  // ← 确保调用了这个
})
```

---

### 情况B: 没有看到任何日志 ❌

**说明**:
- ❌ 后端没有推送消息
- ❌ 或者前端没有正确监听

**排查步骤**:

#### 步骤1: 检查后端日志

查看后端终端，当发送方发消息时，应该看到：
```
info: Received chat:groupMessage from ...
info: Broadcasted group message to group:...
```

**如果没有这些日志**:
- 后端没有监听到事件
- 检查事件名称是否匹配

#### 步骤2: 检查事件名称

**前端发送** (socket.js):
```javascript
socket.emit('groupMessage', data)  // ← 确认是这个名称
```

**后端监听** (socketService.js):
```javascript
socket.on('groupMessage', ...)  // ← 必须完全一致
```

**后端广播**:
```javascript
io.to(`group:${groupId}`).emit('groupMessage', data)  // ← 也必须一致
```

#### 步骤3: 检查房间成员

在后端执行（如果可以访问）:
```javascript
const clients = io.sockets.adapter.rooms.get('group:69d3f89668f596338b0c1930');
console.log('群组成员数:', clients ? clients.size : 0);
```

如果只有 1 个（发送者），说明接收方没加入房间。

## 🛠️ 快速修复方案

### 方案1: 如果后端日志显示已广播，但前端没收到

**可能原因**: 房间名称不匹配

**检查**:
```javascript
// 前端加入的房间
console.log(Array.from(socket.rooms))
// 应该包含: 'group:69d3f89668f596338b0c1930'

// 后端广播的房间
io.to('group:69d3f89668f596338b0c1930')  // ← 格式必须一致
```

**修复**: 确保前后端使用相同的房间名称格式

---

### 方案2: 如果收到了消息但界面没更新

**在 Chat.vue 中添加调试日志**:

```javascript
onGroupMessage((data) => {
  console.log('🔔 收到群聊消息:', data)
  console.log('当前聊天对象:', currentContact.value)
  console.log('消息群组ID:', data.groupId)
  console.log('当前群组ID:', currentContact.value?.id)
  
  const isCurrentChat = currentContact.value && 
    String(data.groupId) === String(currentContact.value.id)
  
  console.log('是否当前聊天:', isCurrentChat)
  
  if (isCurrentChat) {
    console.log('✅ 添加消息到列表')
    addMessage(message)
  } else {
    console.log('❌ 不是当前聊天，忽略')
  }
})
```

### 方案3: 临时强制显示所有消息（仅用于测试）

**在控制台执行**:

import('/src/socket.js').then(module => {

  socket.on('groupMessage', (data) => {
    console.log('🚨 收到消息:', data);
    
    // 强制添加到当前聊天（不考虑是否是当前群组）

    const time = now.getHours() + ':' + (now.getMinutes() < 10 ? '0' : '') + now.getMinutes();

      type: 'received',
      content: data.content,
      time: time,
      senderId: data.sender?._id || data.sender,
      clientMsgId: data.clientMsgId
    };
    
    console.log('强制添加消息:', message);
    
    // 这里需要访问 Vue 组件的 messages 数组
    // 可能需要通过全局变量或其他方式
  });
});
