const DOUYIN_KEY = Buffer.from([0x18, 0x47, 0x70, 0x32, 0x45, 0x11, 0x07, 0x21, 0x09, 0x04, 0x04, 0x05, 0x07, 0x09, 0x02, 0x01]);
const DOUYIN_IV = Buffer.from([0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08, 0x09, 0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06]);

// ===================== 修复你的核心函数 =====================
processDouyinBinaryData(data, host) {
    // 1. 修复：头部是8字节，不是4字节
    if (data.length < 8) return;

    // 2. 解析标准头部
    const opcode = data.readUInt32BE(0);
    const dataLength = data.readUInt32BE(4);
    // 3. 精准截取加密数据（避免粘包）
    const encryptedData = data.slice(8, 8 + dataLength);

    console.log(`抖音数据包：opcode=${opcode}, 数据长度=${dataLength}`);

    try {
        // 核心步骤：AES解密
        const decryptedData = this.aesDecrypt(encryptedData);
        // 解析ProtoBuf数据
        this.parseProtoData(opcode, decryptedData);
    } catch (err) {
        console.error('解密/解析失败：', err.message);
    }
}

// AES-128-CBC 解密函数
aesDecrypt(encryptedData) {
    const decipher = crypto.createDecipheriv('aes-128-cbc', DOUYIN_KEY, DOUYIN_IV);
    decipher.setAutoPadding(true);
    let decrypted = decipher.update(encryptedData);
    decrypted = Buffer.concat([decrypted, decipher.final()]);
    return decrypted;
}

// ProtoBuf解析 + 数据推送
parseProtoData(opcode, data) {
    let result;
    let type;

    switch (opcode) {
        case 0x0001:
            // 解析弹幕
            result = DanmakuMsg.decode(data);
            type = 'danmaku';
            console.log('✅ 弹幕解析：', result);
            break;
        case 0x0002:
            // 解析点赞
            result = LikeMsg.decode(data);
            type = 'like';
            console.log('✅ 点赞解析：', result);
            break;
        default:
            return;
    }

    // 实时推送给前端WebSocket
    this.pushToClient({ type, data: result.toJSON() });
}

// WebSocket实时推送（你需要提前建立WS服务）
pushToClient(msg) {
    if (this.wss && this.wss.clients) {
        this.wss.clients.forEach(client => {
            if (client.readyState === WebSocket.OPEN) {
                client.send(JSON.stringify(msg));
            }
        });
    }
}
