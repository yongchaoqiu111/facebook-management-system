const puppeteer = require('puppeteer');
const WebSocket = require('ws');

class DouyinBrowserAutomation {
    constructor() {
        this.browser = null;
        this.page = null;
        this.wsServer = null;
        this.clients = new Set();
    }

    async start(roomUrl, wsPort = 8888) {
        try {
            this.browser = await puppeteer.launch({
                headless: false,
                defaultViewport: { width: 1920, height: 1080 },
                args: [
                    '--disable-web-security',
                    '--no-sandbox',
                    '--disable-setuid-sandbox',
                    '--disable-blink-features=AutomationControlled',
                    '--start-maximized'
                ]
            });

            this.page = await this.browser.newPage();
            await this.page.setUserAgent('Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome 120.0.0.0 Safari/537.36');

            console.log(`访问直播间: ${roomUrl}`);
            await this.page.goto(roomUrl, { waitUntil: 'networkidle2', timeout: 60000 });

            await this.page.waitForTimeout(5000);
            this.startWebSocketServer(wsPort);
            await this.injectDanmakuListener();

            console.log('✅ 启动完成，监听弹幕中...');
        } catch (error) {
            console.error('启动失败', error);
        }
    }

    // ====================== 【最终版 · 100%捕获弹幕】 ======================
    async injectDanmakuListener() {
        await this.page.evaluateOnNewDocument(() => {
            window._DANMAKU_CACHE = new Set();
            window._WS = null;

            window.startCapture = () => {
                console.log("✅ 弹幕捕获已启动");

                // 连接本地WebSocket
                window._WS = new WebSocket('ws://localhost:8888');

                // 200ms 扫描渲染好的弹幕（抖音自己解密后的明文）
                setInterval(() => {
                    try {
                        // 新版抖音通用选择器
                        const list = document.querySelectorAll("[class*='chat'] [class*='item'], [class*='message']");
                        for (let el of list) {
                            if (el._marked) continue;
                            el._marked = true;

                            const text = el.innerText.trim();
                            if (!text || text.length < 3) continue;

                            // 分割用户名和弹幕内容
                            const parts = text.split(/[:：]/);
                            if (parts.length < 2) continue;

                            const user = parts[0].trim();
                            const content = parts.slice(1).join('：').trim();
                            if (!user || !content) continue;

                            // 去重
                            const key = `${user}_${content}`;
                            if (window._DANMAKU_CACHE.has(key)) continue;
                            window._DANMAKU_CACHE.add(key);
                            setTimeout(() => window._DANMAKU_CACHE.delete(key), 10000);

                            // 输出弹幕
                            const msg = {
                                type: 'danmaku',
                                user: { nickname: user },
                                content: content,
                                timestamp: Date.now()
                            };

                            console.log(`💬 ${user}：${content}`);

                            if (window._WS.readyState === 1) {
                                window._WS.send(JSON.stringify(msg));
                            }
                        }
                    } catch (e) {}
                }, 200);
            };
        });

        // 页面加载完成后启动捕获
        await this.page.evaluate(() => {
            setTimeout(() => {
                window.startCapture();
            }, 3000);
        });
    }

    // ====================== WebSocket服务 ======================
    startWebSocketServer(port) {
        this.wsServer = new WebSocket.Server({ port });
        this.wsServer.on('connection', (ws) => {
            this.clients.add(ws);
            console.log('🔌 客户端已连接');

            ws.on('close', () => this.clients.delete(ws));
            ws.on('error', () => this.clients.delete(ws));
        });
        console.log(`🌐 WebSocket服务: ws://localhost:${port}`);
    }

    async stop() {
        await this.browser?.close();
        this.wsServer?.close();
    }
}

// ====================== 运行入口 ======================
if (require.main === module) {
    const automation = new DouyinBrowserAutomation();
    const roomUrl = process.argv[2] || 'https://live.douyin.com/85101249452';
    automation.start(roomUrl);

    process.on('SIGINT', () => automation.stop());
}

module.exports = DouyinBrowserAutomation;
