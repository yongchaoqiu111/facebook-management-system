### Step 1: 添加消息去重逻辑（30 分钟）

**文件**：`src/composables/useMessageCenter.js`

在 `useMessageCenter` 函数中添加：

```javascript
export function useMessageCenter() {
  // ... 现有代码
  
  // ✅ 添加消息去重 Set
  const receivedMsgIds = new Set()
  const MAX_STORED_IDS = 1000  // 最多存储 1000 个 ID
  
  /**
   * 检查消息是否重复
   */
  const isDuplicateMessage = (message) => {
    const msgId = message._id || message.clientMsgId
    if (!msgId) return false
    
    if (receivedMsgIds.has(msgId)) {
      console.log('⚠️ 检测到重复消息:', msgId)
      return true
    }
    
    receivedMsgIds.add(msgId)
    
    // 限制 Set 大小，避免内存泄漏
    if (receivedMsgIds.size > MAX_STORED_IDS) {
      // 清理最早的 500 个 ID
      const idsArray = Array.from(receivedMsgIds)
      idsArray.slice(0, 500).forEach(id => receivedMsgIds.delete(id))
      console.log('🗑️ 清理了 500 个旧消息 ID')
    }
    
    return false
  }
  
  /**
   * 添加消息到对话数组（带去重）
   */
  const addMessageToConversation = (conversationId, message) => {
    // ✅ 检查是否重复
    if (isDuplicateMessage(message)) {
      return  // 重复消息，直接返回
    }
    
    // 初始化对话数组
    if (!conversations[conversationId]) {
      conversations[conversationId] = []
    }
    
    // 使用 splice 保持响应式
    conversations[conversationId].splice(
      conversations[conversationId].length,
      0,
      message
    )
  }
  
  // ... 其他代码
  
  return {
    // ... 现有返回值
    isDuplicateMessage,  // ✅ 暴露出去（可选，用于调试）
    receivedMsgIds       // ✅ 暴露出去（可选，用于调试）
  }
}
```

**关键点**：
- 使用 `Set` 存储已接收的消息 ID
- 优先使用 `_id`（服务端生成），其次使用 `clientMsgId`
- 限制 Set 大小为 1000，避免内存泄漏
- 超过限制时清理最早的 500 个 ID

**检查点**：
- [ ] 发送同一条消息两次，只显示一次
- [ ] 刷新页面后，Set 清空（正常行为）
