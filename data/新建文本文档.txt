1. Cookie 处理问题（直接给你稳定解析）
问题
txt 格式 cookie 解析失败 → 登录失效。
解决方案（通用、最稳）
js
// 一行解析  netscape / 普通txt cookie
function parseCookieString(cookieStr) {
  return cookieStr
    .split('\n')
    .map(line => line.trim())
    .filter(line => line && !line.startsWith('#') && !line.startsWith('Domain'))
    .map(line => {
      const parts = line.split('\t');
      if (parts.length < 7) return null;
      return {
        domain: parts[0],
        path: parts[2],
        secure: parts[3] === 'TRUE',
        expires: parseInt(parts[4]) || 0,
        name: parts[5],
        value: parts[6]
      };
    })
    .filter(Boolean);
}
js
// 加载
const cookies = parseCookieString(fs.readFileSync('cookie.txt','utf8'))
await page.setCookie(...cookies);

7. 异常处理（完整兜底）
js
async function safeAction(fn, retries = 2) {
  for (let i = 0; i < retries; i++) {
    try {
      return await fn();
    } catch (e) {
      console.log(`重试 ${i+1}`, e.message);
      await page.waitForTimeout(2000);
    }
  }
  return null;
}
使用：
js
await safeAction(() => postElement.click());
await safeAction(() => page.type(...));
